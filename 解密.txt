import javax.crypto.Cipher;
import javax.crypto.spec.SecretKeySpec;
import java.util.Base64;

public class DecryptionExample {
    public static void main(String[] args) throws Exception {
        String encodedText = "YourEncodedTextHere"; // 替换为您的编码文本
        String secretKey = "MySecretKey12345"; // 替换为您自己的密钥

        // 创建一个用于加密算法（例如AES）的SecretKeySpec
        SecretKeySpec secretKeySpec = new SecretKeySpec(secretKey.getBytes("UTF-8"), "AES");

        // 创建一个用于解密的Cipher对象
        Cipher cipher = Cipher.getInstance("AES");
        cipher.init(Cipher.DECRYPT_MODE, secretKeySpec);

        // 解码Base64编码的文本
        byte[] decodedText = Base64.getDecoder().decode(encodedText);

        // 解密文本
        byte[] decryptedText = cipher.doFinal(decodedText);

        System.out.println("解密后的文本: " + new String(decryptedText, "UTF-8"));
    }
}
